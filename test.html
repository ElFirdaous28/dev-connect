/* Exemple d'optimisation avec index */
SELECT u.username, p.title 
FROM users u
JOIN posts p ON u.id = p.user_id
WHERE p.created_at > '2023-01-01'
AND u.active = true
ORDER BY p.created_at DESC
LIMIT 20;
